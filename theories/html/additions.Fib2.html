Definition mul2 (p q : N × N) :=
  match p, q with
    (a, b),(c,d) ⇒ (a×c + a×d + b×c, a×c + b×d)
  end.

Lemma neutral_l p : mul2 (0,1) p = p.

Lemma neutral_r p : mul2 p (0,1) = p.

#[ global ] Instance Mul2 : Monoid mul2 (0,1).
