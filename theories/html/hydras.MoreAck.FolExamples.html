Module Toy.
  Inductive Rel: Set := A_ | B_ | C_ | P_ | Q_ | R_.
  Inductive Fun : Set := a_ | b_ | f_ | g_ | h_.

  Definition arityR (x : Rel): nat :=
    match x with
       P_ | Q_ ⇒ 1 | R_ ⇒ 2 | _ ⇒ 0
    end.

 Definition arityF (x : Fun): nat :=
    match x with f_ | g_ ⇒ 1 | h_ ⇒ 2 | _ ⇒ 0 end.

 Definition L := language Rel Fun arityR arityF.
