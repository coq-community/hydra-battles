    ∀ f, (∀ k, f (S k) = S k) → L_spec zero f

Definition largeb (alpha : T1) (s: list nat) :=
  match (gnaw alpha s)
          with zero ⇒ true | _ ⇒ false end.

  largeb alpha s.

  largeSb alpha s.
