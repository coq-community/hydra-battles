Ltac old_rename := rename_hyp_default.

Ltac rename_short n th :=
  match th with
  | (?f ?x ?y) ⇒ name ((f # 1) ++ (x # 1))
  | (?f ?x) ⇒ name ((f # 1))
  | _ ⇒ old_rename n th
  end.
